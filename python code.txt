import os
import subprocess
import json
import shutil
import re
from google.colab import files
import ipywidgets as widgets
from IPython.display import display

DOWNLOAD_DIR = "downloads"

# ---------------- CLEAN DOWNLOAD FOLDER ----------------
def clean_downloads():
    if os.path.exists(DOWNLOAD_DIR):
        shutil.rmtree(DOWNLOAD_DIR)
    os.makedirs(DOWNLOAD_DIR, exist_ok=True)

# ---------------- UI ELEMENTS ----------------
url_input = widgets.Text(
    placeholder="Paste YouTube video URL here",
    description="YouTube URL:",
    layout=widgets.Layout(width="80%")
)

resolution_dropdown = widgets.Dropdown(
    description="Resolution:",
    disabled=True
)

progress_bar = widgets.IntProgress(
    value=0,
    min=0,
    max=100,
    description="Download:",
    bar_style=""
)

progress_label = widgets.Label(value="0%")
status_label = widgets.Label()

fetch_button = widgets.Button(
    description="Fetch Resolutions",
    button_style="info"
)

download_button = widgets.Button(
    description="Download Video",
    button_style="success",
    disabled=True
)

# ---------------- HELPERS ----------------
def format_size(bytes_size):
    return f"{bytes_size / (1024*1024):.2f} MB"

# ---------------- FETCH RESOLUTIONS ----------------
def fetch_resolutions(b):
    clean_downloads()
    download_button.disabled = True
    resolution_dropdown.disabled = True
    progress_bar.value = 0
    progress_label.value = "0%"
    status_label.value = ""

    video_url = url_input.value.strip()
    if not video_url:
        status_label.value = "❌ Please paste a YouTube URL"
        return

    status_label.value = "⏳ Fetching available resolutions..."

    cmd = ["yt-dlp", "-J", video_url]
    result = subprocess.run(cmd, capture_output=True, text=True)

    if not result.stdout:
        status_label.value = "❌ Unable to fetch video info (restricted or unavailable)"
        return

    info = json.loads(result.stdout)

    # Get best available audio size
    audio_size = 0
    for f in info.get("formats", []):
        if f.get("acodec") != "none" and f.get("vcodec") == "none":
            audio_size = max(audio_size, f.get("filesize") or f.get("filesize_approx") or 0)

    resolutions = {}

    for f in info.get("formats", []):
        height = f.get("height")
        if f.get("vcodec") != "none" and height:
            video_size = f.get("filesize") or f.get("filesize_approx")
            resolutions[height] = (video_size or 0) + audio_size

    STANDARD_RES = [144, 240, 360, 480, 720, 1080, 1440, 2160]

    options = []
    for r in STANDARD_RES:
        if r in resolutions:
            size = resolutions[r]
            size_text = format_size(size) if size > 0 else "Size unknown"
            options.append((f"{r}p — {size_text}", r))

    if not options:
        options = [("Best available quality", 9999)]
        status_label.value = "⚠️ Limited metadata. Using best available quality."
    else:
        status_label.value = "✅ Select resolution and click Download"

    resolution_dropdown.options = options
    resolution_dropdown.value = options[0][1]
    resolution_dropdown.disabled = False
    download_button.disabled = False

# ---------------- DOWNLOAD WITH % PROGRESS ----------------
def download_video(b):
    video_url = url_input.value.strip()
    selected_resolution = resolution_dropdown.value

    download_button.disabled = True
    resolution_dropdown.disabled = True
    progress_bar.value = 0
    progress_label.value = "0%"
    status_label.value = "⬇️ Downloading video..."

    command = [
        "yt-dlp",
        "-f", f"bv*[height<={selected_resolution}]+ba/b",
        "--merge-output-format", "mp4",
        "--newline",
        "-o", f"{DOWNLOAD_DIR}/%(title)s.%(ext)s",
        video_url
    ]

    process = subprocess.Popen(
        command,
        stdout=subprocess.PIPE,
        stderr=subprocess.STDOUT,
        text=True
    )

    for line in process.stdout:
        match = re.search(r'(\d{1,3}\.\d+)%', line)
        if match:
            percent = int(float(match.group(1)))
            progress_bar.value = percent
            progress_label.value = f"{percent}%"

    process.wait()

    status_label.value = "✅ Download complete!"

    for file in os.listdir(DOWNLOAD_DIR):
        if file.endswith(".mp4"):
            files.download(os.path.join(DOWNLOAD_DIR, file))

# ---------------- BUTTON EVENTS ----------------
fetch_button.on_click(fetch_resolutions)
download_button.on_click(download_video)

display(
    url_input,
    fetch_button,
    resolution_dropdown,
    download_button,
    progress_bar,
    progress_label,
    status_label
)
